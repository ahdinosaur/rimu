# List

- "a"
- "b"
- "c"

==>

Template(BlockList(String, String, String))

# Nested List

- - - "a"
  - "b"
- "c"

==>

Template(
  BlockList(
    BlockList(
      BlockList(String)
      String,
    ),
    String
  )
)

# Object (identifier keys, string values)

a: "b"
c: "d"

==>

Template(
  BlockObject(
    BlockEntry(
      BlockKey(Identifier),
      String
    ),
    BlockEntry(
      BlockKey(Identifier),
      String
    )
  )
)

# Object with a list value

a:
  - "b"
  - "c"
d: "e"

==>

Template(
  BlockObject(
    BlockEntry(
      BlockKey(Identifier),
      BlockWrap(
        BlockList(
          String,
          String
        )
      )
    ),
    BlockEntry(
      BlockKey(Identifier),
      String
    )
  )
)
